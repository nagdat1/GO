//Truncate Function
truncate(number, decimals) =>
    factor = math.pow(10, decimals)
    int(number * factor) / factor

// Format Time Function - Convert timestamp to "yyyy-MM-dd HH:mm" format
f_formatTime(_time) =>
    _year = str.tostring(year(_time))
    _month = str.tostring(month(_time))
    _month := str.length(_month) == 1 ? "0" + _month : _month
    _day = str.tostring(dayofmonth(_time))
    _day := str.length(_day) == 1 ? "0" + _day : _day
    _hour = str.tostring(hour(_time))
    _hour := str.length(_hour) == 1 ? "0" + _hour : _hour
    _minute = str.tostring(minute(_time))
    _minute := str.length(_minute) == 1 ? "0" + _minute : _minute
    _year + "-" + _month + "-" + _day + " " + _hour + ":" + _minute

    _tpLine    := _condition[1] !=  _conditionValue and _leTrigger ? _topLvl :
                  _condition[1] != -_conditionValue and _seTrigger ? _botLvl :
                  nz(_tpLine[1])
    [_tpLine]

                       _scr1 < _scr2 and _scr1[1] > _scr2[1]
